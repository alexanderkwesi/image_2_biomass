# auth-services/app/main.py
from pydantic import BaseModel, EmailStr, validator, root_validator
from typing import Optional, Dict, Any
from datetime import datetime

class UserRegister(BaseModel):
    first_name: str
    last_name: str
    email: EmailStr
    password: str
    confirm_password: str
    
    @validator('first_name', 'last_name')
    def name_must_not_be_empty(cls, v):
        if not v.strip():
            raise ValueError('Name cannot be empty')
        return v.strip()
    
    @validator('password')
    def password_must_be_strong(cls, v):
        if len(v) < 6:
            raise ValueError('Password must be at least 6 characters long')
        return v
    
    @root_validator
    def passwords_match(cls, values):
        password = values.get('password')
        confirm_password = values.get('confirm_password')
        if password and confirm_password and password != confirm_password:
            raise ValueError('Passwords do not match')
        return values

class UserLogin(BaseModel):
    email: EmailStr
    password: str

class UserResponse(BaseModel):
    id: str
    first_name: str
    last_name: str
    email: str
    is_verified: bool
    created_at: datetime

class AuthResponse(BaseModel):
    success: bool
    message: str
    access_token: Optional[str] = None
    token_type: Optional[str] = None
    user: Optional[UserResponse] = None

class ErrorResponse(BaseModel):
    success: bool
    detail: str

class TokenData(BaseModel):
    user_id: Optional[str] = None
    email: Optional[str] = None

class TokenVerify(BaseModel):
    token: str

from pydantic import BaseModel
from typing import Optional, Dict, Any, List
from datetime import datetime
from enum import Enum

class PredictionType(str, Enum):
    BIOMASS_ESTIMATION = "biomass_estimation"
    PLANT_HEALTH = "plant_health"
    SPECIES_DETECTION = "species_detection"

class PredictionRequest(BaseModel):
    user_id: str
    image_id: str
    prediction_type: PredictionType
    model_parameters: Optional[Dict[str, Any]] = None

class PredictionResponse(BaseModel):
    success: bool
    prediction_id: str
    prediction_type: str
    prediction_data: Dict[str, Any]
    confidence_score: Optional[float] = None
    model_version: str
    processing_time_ms: int
    status: str

class BiomassEstimationData(BaseModel):
    biomass_kg_per_hectare: float
    vegetation_coverage: float
    plant_height_cm: Optional[float] = None
    confidence_interval: Dict[str, float]

class PlantHealthData(BaseModel):

    nutrient_deficiencies: List[str]

class SpeciesDetectionData(BaseModel):
